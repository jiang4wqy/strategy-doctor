# Strategy Doctor 开发实施计划（Implementation Plan）

> 状态说明：这是历史实施计划，任务已完成。当前采用 snapshot builder + `search.ts`，准确接口和命令见根目录 `README.md`。

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建"AI 策略体检与优化引擎"最短闭环 MVP——输入任意参数化策略，自动用 5 维对抗场景压测出死因，按 3 种投资风格评分，开出定向处方，并在分离的 held-out 场景集上复测、输出诚实的"稳健性↔收益"取舍。

**Architecture:** 六层流水线：接口契约 → 回测适配层（Mock 为主线 / Bitget Agent Hub 为增强）→ 红队对抗层（5 个 Bitget Skill = 5 攻击维度，参数化场景 + 对抗搜索）→ 死因诊断 → 三风格评分 → 处方进化（死因驱动的定向变异 + 邻域搜索）+ held-out 复测。全程 seed 驱动、确定性可复现，demo 离线可跑。

**Tech Stack:** Node 24 原生 TypeScript（类型擦除直接运行，零构建）、node:test + node:assert（零测试依赖）、Anthropic API（死亡报告叙事，模板兜底）、Bitget Agent Hub（`npx bitget-hub`，spike 验证后接入）。

```ts
// src/backtest/path.ts — 确定性随机源 + 场景→价格路径
import type { MarketShock } from '../contracts.ts';

/** 经典 mulberry32：同 seed 同序列，跨平台一致 */
export function mulberry32(seed: number): () => number {
  let a = seed >>> 0;
  return () => {
    a = (a + 0x6d2b79f5) | 0;
    let t = Math.imul(a ^ (a >>> 15), 1 | a);
    t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t;
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}

/**
 * 把 MarketShock 翻译成价格序列。
 * 前 1/3 为平静期（微弱上行，诱使趋势策略入场做多），随后注入冲击。
 */
export function generatePath(shock: MarketShock, totalBars = 240, startPrice = 100): number[] {
  const rng = mulberry32(shock.seed);
  const baseVol = 0.008 * shock.volMult;
  const prices = [startPrice];
  const t0 = Math.floor(totalBars / 3);
  const t1 = Math.min(totalBars - 1, t0 + shock.durationBars);
  const span = Math.max(1, t1 - t0);

  for (let i = 1; i < totalBars; i++) {
    let drift = 0.0005; // 平静期微弱上行
    let jump = 0;
    if (i >= t0 && i < t1) {
      switch (shock.kind) {
        case 'crash':
        case 'grind': // grind 与 crash 同形，靠更长 durationBars 体现"阴跌"
          drift = -shock.magnitude / span;
          break;
        case 'squeeze': // 先逼空上冲 1/3 时长，再加倍下杀
          drift = i < t0 + span / 3
            ? (shock.magnitude / 2) / (span / 3)
            : -(1.5 * shock.magnitude) / ((2 * span) / 3);
          break;
        case 'whipsaw': // 正弦震荡，无净方向
          drift = (Math.sin(((i - t0) / span) * Math.PI * 6) * shock.magnitude) / 8;
          break;
        case 'gap': // 单 bar 跳空
          if (i === t0) jump = -shock.magnitude;
          break;
      }
    }
    const noise = (rng() * 2 - 1) * baseVol;
    prices.push(Math.max(0.01, prices[i - 1] * (1 + drift + noise + jump)));
  }
  return prices;
}
```

```ts
// src/backtest/mock.ts — 确定性 Mock 回测引擎
import type { BacktestAdapter, Metrics, Scenario, Strategy, StrategyParams } from '../contracts.ts';
import { generatePath } from './path.ts';

function sma(prices: number[], period: number, i: number): number | null {
  if (i + 1 < period) return null;
  let s = 0;
  for (let k = i - period + 1; k <= i; k++) s += prices[k];
  return s / period;
}

/** 纯函数核心：给定参数与价格序列，逐 bar 模拟。便于单测。 */
export function runOnPrices(params: StrategyParams, prices: number[]): Metrics {
  const { fastMA, slowMA, leverage, stopLossPct, positionPct } = params;
  let equity = 1;
  const equityCurve = [equity];
  let dir = 0;          // 1 多 / -1 空 / 0 空仓
  let entry = 0;
  let blockedDir = 0;   // 止损后封禁的方向，信号翻转才解除
  let liquidated = false;
  let numTrades = 0;
  let peak = 1;
  let maxDD = 0;
  const liqLine = 0.9 / leverage; // 爆仓线：不利偏移≥90%保证金

  for (let i = 1; i < prices.length; i++) {
    // 1) 持仓结算本 bar 盈亏（单 bar 最多亏掉本仓保证金）
    if (dir !== 0 && !liquidated) {
      const barRet = dir * (prices[i] / prices[i - 1] - 1);
      const dEq = Math.max(-positionPct, barRet * leverage * positionPct);
      equity = Math.max(0.001, equity * (1 + dEq));
      const excursion = dir * (prices[i] / entry - 1);
      if (excursion <= -liqLine) {
        liquidated = true;          // 爆仓：仓位清零
        blockedDir = dir; dir = 0;
      } else if (excursion <= -stopLossPct) {
        blockedDir = dir; dir = 0;  // 止损离场，等信号翻转再进
      }
    }
    if (equity <= 0.05) { liquidated = true; dir = 0; } // 权益地板=视同爆仓

    // 2) 信号与开仓/翻仓
    const fast = sma(prices, fastMA, i);
    const slow = sma(prices, slowMA, i);
    const signal = fast !== null && slow !== null ? Math.sign(fast - slow) : 0;
    if (!liquidated && signal !== 0 && signal !== dir && signal !== blockedDir) {
      dir = signal; entry = prices[i]; numTrades++; blockedDir = 0;
    }

    peak = Math.max(peak, equity);
    maxDD = Math.max(maxDD, (peak - equity) / peak);
    equityCurve.push(equity);
  }
  return { pnlPct: equity - 1, maxDrawdownPct: maxDD, liquidated, numTrades, equityCurve };
}

export class MockBacktester implements BacktestAdapter {
  async run(strategy: Strategy, scenario: Scenario): Promise<Metrics> {
    return runOnPrices(strategy.params, generatePath(scenario.shock));
  }
}
```

```ts
// src/redteam/sample.ts — 从场景族确定性采样具体场景
import type { Scenario } from '../contracts.ts';
import { mulberry32 } from '../backtest/path.ts';
import type { ScenarioFamily } from './templates.ts';

export function sampleScenario(family: ScenarioFamily, rng: () => number, id: string): Scenario {
  const lerp = (r: [number, number]) => r[0] + rng() * (r[1] - r[0]);
  const magnitude = lerp(family.magnitudeRange);
  return {
    id,
    name: family.name,
    dimension: family.dimension,
    sourceSkill: family.sourceSkill,
    severity: Math.min(5, Math.max(1, Math.round(magnitude * 10))),
    narrative: family.narrativeTemplate.replace('{magnitude}', (magnitude * 100).toFixed(0)),
    shock: {
      kind: family.kind,
      magnitude,
      durationBars: Math.round(lerp(family.durationRange)),
      volMult: lerp(family.volMultRange),
      seed: Math.floor(rng() * 2 ** 31),
    },
  };
}

/** 治疗集与 held-out 集都用本函数生成，靠不同 seed 保证分离 */
export function sampleScenarioSet(families: ScenarioFamily[], perFamily: number, seed: number, idPrefix: string): Scenario[] {
  const rng = mulberry32(seed);
  const out: Scenario[] = [];
  for (const f of families) {
    for (let i = 0; i < perFamily; i++) out.push(sampleScenario(f, rng, `${idPrefix}-${f.dimension}-${i}`));
  }
  return out;
}
```

```ts
// src/redteam/search.ts — 对抗搜索：在采样场景中挑最毒
import type { Metrics, Scenario } from '../contracts.ts';

export interface ScenarioRun {
  scenario: Scenario;
  metrics: Metrics;
}

/** 伤害分：清算压倒一切，其次回撤，最后亏损 */
export function damageScore(m: Metrics): number {
  return (m.liquidated ? 1000 : 0) + m.maxDrawdownPct * 100 - m.pnlPct * 100;
}

/** 每个维度保留伤害最大的一个场景，整体按伤害降序 */
export function pickWorstPerDimension(runs: ScenarioRun[]): ScenarioRun[] {
  const byDim = new Map<string, ScenarioRun>();
  for (const r of runs) {
    const cur = byDim.get(r.scenario.dimension);
    if (!cur || damageScore(r.metrics) > damageScore(cur.metrics)) byDim.set(r.scenario.dimension, r);
  }
  return [...byDim.values()].sort((a, b) => damageScore(b.metrics) - damageScore(a.metrics));
}
```

```ts
// src/scoring/scorecard.ts — 一组回测结果 → 单风格评分
import type { Metrics, StyleScore } from '../contracts.ts';
import type { StyleProfile } from './styles.ts';

export function scoreStyle(results: Metrics[], profile: StyleProfile): StyleScore {
  const n = results.length;
  const liqRate = results.filter(r => r.liquidated).length / n;
  // 清算场景由 liquidationTolerance 单独问责；回撤阈值只衡量未清算场景，
  // 否则"激进型容忍 20% 清算"会被清算场景自带的 ~100% 回撤直接判死，容忍度形同虚设。
  const survivors = results.filter(r => !r.liquidated);
  const scoringDD = survivors.length > 0 ? Math.max(...survivors.map(r => r.maxDrawdownPct)) : 1;
  const worstDD = Math.max(...results.map(r => r.maxDrawdownPct)); // 报告展示用：全量最差（诚实）
  const meanPnl = results.reduce((s, r) => s + r.pnlPct, 0) / n;

  const ddScore = Math.max(0, 1 - scoringDD / profile.maxDrawdown);     // 回撤逼近阈值→0
  const pnlScore = Math.min(1, Math.max(0, 0.5 + meanPnl));             // -50%→0 / +50%→1
  const liqPenalty = liqRate > profile.liquidationTolerance ? 0.3 : 1;  // 超容忍度重罚

  return {
    style: profile.style,
    riskScore: Math.round(100 * (profile.ddWeight * ddScore + profile.pnlWeight * pnlScore) * liqPenalty),
    survived: scoringDD <= profile.maxDrawdown && liqRate <= profile.liquidationTolerance,
    worstDrawdownPct: worstDD,
    meanPnlPct: meanPnl,
  };
}
```

```ts
// src/prescribe/mutations.ts — 死因→定向修补 + 邻域抖动
import type { DeathCause, StrategyParams } from '../contracts.ts';

/** 每条死因对应一组明确修补；rationale 与 patch 一一对应（报告可解释性） */
export function targetedPatch(params: StrategyParams, causes: DeathCause[]): { patch: Partial<StrategyParams>; rationale: string[] } {
  const patch: Partial<StrategyParams> = {};
  const rationale: string[] = [];
  const set = new Set(causes);

  if (set.has('liquidation')) {
    patch.leverage = Math.max(1, Math.round(params.leverage / 2));
    patch.stopLossPct = Math.min(params.stopLossPct, +(0.8 / patch.leverage / 2).toFixed(3));
    rationale.push(`清算死因 → 杠杆 ${params.leverage}→${patch.leverage}，止损收紧到爆仓线一半以内`);
  }
  if (set.has('drawdown-breach')) {
    patch.positionPct = +(params.positionPct * 0.7).toFixed(2);
    rationale.push(`回撤击穿 → 仓位 ${params.positionPct}→${patch.positionPct}`);
  }
  if (set.has('stop-loss-bleed')) {
    patch.fastMA = Math.round(params.fastMA * 1.5);
    patch.slowMA = Math.round(params.slowMA * 1.5);
    rationale.push(`震荡反复止损放血 → 均线周期放慢 1.5 倍过滤噪音`);
  }
  return { patch, rationale };
}

/** 围绕基准参数 ±20% 抖动，生成邻域候选（合法性钳制） */
export function jitterParams(p: StrategyParams, rng: () => number): StrategyParams {
  const j = (v: number) => v * (0.8 + rng() * 0.4);
  const fastMA = Math.max(2, Math.round(j(p.fastMA)));
  return {
    fastMA,
    slowMA: Math.max(fastMA + 2, Math.round(j(p.slowMA))),
    leverage: Math.max(1, Math.round(j(p.leverage))),
    stopLossPct: Math.min(0.99, Math.max(0.01, +j(p.stopLossPct).toFixed(3))),
    positionPct: Math.min(1, Math.max(0.1, +j(p.positionPct).toFixed(2))),
  };
}

export function diffParams(before: StrategyParams, after: StrategyParams): Partial<StrategyParams> {
  const out: Partial<StrategyParams> = {};
  for (const k of Object.keys(after) as (keyof StrategyParams)[]) {
    if (before[k] !== after[k]) out[k] = after[k];
  }
  return out;
}
```

```ts
// src/prescribe/evolve.ts — 处方：定向修补为基准 + 邻域搜索择优
import type { BacktestAdapter, Death, Metrics, Prescription, Scenario, Strategy, StrategyParams } from '../contracts.ts';
import { mulberry32 } from '../backtest/path.ts';
import { scoreStyle } from '../scoring/scorecard.ts';
import type { StyleProfile } from '../scoring/styles.ts';
import { diffParams, jitterParams, targetedPatch } from './mutations.ts';

export interface PrescribeOptions {
  candidates: number;  // 候选数（含定向基准）
  seed: number;
}

export async function prescribe(
  strategy: Strategy,
  deaths: Death[],
  treatment: Scenario[],
  backtest: BacktestAdapter,
  profile: StyleProfile,
  opts: PrescribeOptions = { candidates: 8, seed: 7 },
): Promise<Prescription> {
  const causes = deaths.map(d => d.cause).filter(c => c !== 'survived');
  const { patch, rationale } = targetedPatch(strategy.params, causes);

  // 候选 = 定向修补基准 + 围绕基准的邻域抖动
  const rng = mulberry32(opts.seed);
  const base: StrategyParams = { ...strategy.params, ...patch };
  const candidates: StrategyParams[] = [base];
  for (let i = 1; i < opts.candidates; i++) candidates.push(jitterParams(base, rng));

  // 择优目标：所选风格在治疗集上的风险分（杠杆不得回升超过基准，守住定向修补的意图）
  let best = base;
  let bestScore = -Infinity;
  for (const cand of candidates) {
    if (cand.leverage > base.leverage) continue;
    const trial: Strategy = { ...strategy, id: `${strategy.id}-rx`, params: cand };
    const results: Metrics[] = [];
    for (const sc of treatment) results.push(await backtest.run(trial, sc));
    const score = scoreStyle(results, profile).riskScore;
    if (score > bestScore) { bestScore = score; best = cand; }
  }

  return {
    changes: diffParams(strategy.params, best),
    rationale: rationale.join('；'),
    patchedStrategy: { ...strategy, id: `${strategy.id}-rx`, name: `${strategy.name}（处方修补版）`, params: best },
  };
}
```

```ts
// src/prescribe/validate.ts — held-out 复测：原版 vs 处方版的诚实对比
import type { BacktestAdapter, Metrics, Scenario, Strategy, Tradeoff } from '../contracts.ts';
import { scoreStyle } from '../scoring/scorecard.ts';
import type { StyleProfile } from '../scoring/styles.ts';

export async function validateOnHeldOut(
  original: Strategy,
  patched: Strategy,
  heldOut: Scenario[],
  backtest: BacktestAdapter,
  profile: StyleProfile,
): Promise<Tradeoff> {
  const runAll = async (s: Strategy): Promise<Metrics[]> => {
    const out: Metrics[] = [];
    for (const sc of heldOut) out.push(await backtest.run(s, sc));
    return out;
  };
  const [origResults, rxResults] = [await runAll(original), await runAll(patched)];
  const origScore = scoreStyle(origResults, profile);
  const rxScore = scoreStyle(rxResults, profile);
  return {
    robustnessGain: rxScore.riskScore - origScore.riskScore,
    returnCost: +(rxScore.meanPnlPct - origScore.meanPnlPct).toFixed(4),
  };
}
```

```ts
// src/pipeline/doctor.ts — 体检总编排：诊断→评分→处方→held-out 复测
import type { BacktestAdapter, Death, Prescription, Scorecard, Strategy, StyleName, StyleScore, Tradeoff } from '../contracts.ts';
import { sampleScenarioSet } from '../redteam/sample.ts';
import { FAMILIES } from '../redteam/templates.ts';
import { pickWorstPerDimension, type ScenarioRun } from '../redteam/search.ts';
import { classifyDeath } from '../redteam/diagnose.ts';
import { STYLES, getProfile } from '../scoring/styles.ts';
import { scoreStyle } from '../scoring/scorecard.ts';
import { prescribe } from '../prescribe/evolve.ts';
import { validateOnHeldOut } from '../prescribe/validate.ts';

export interface DoctorOptions {
  style: StyleName;             // 处方面向的风格
  treatmentSeed: number;        // 治疗集种子
  heldOutSeed: number;          // 验证集种子（必须 ≠ treatmentSeed）
  scenariosPerDimension: number;
}

export async function runDoctor(strategy: Strategy, backtest: BacktestAdapter, opts: DoctorOptions): Promise<Scorecard> {
  if (opts.treatmentSeed === opts.heldOutSeed) {
    throw new Error('红线：治疗集与验证集种子必须不同（防自我过拟合）');
  }
  const treatment = sampleScenarioSet(FAMILIES, opts.scenariosPerDimension, opts.treatmentSeed, 'tx');
  const heldOut = sampleScenarioSet(FAMILIES, opts.scenariosPerDimension, opts.heldOutSeed, 'ho');

  // 1) 全量压测
  const runs: ScenarioRun[] = [];
  for (const sc of treatment) runs.push({ scenario: sc, metrics: await backtest.run(strategy, sc) });

  // 2) 每维度最毒场景 → 死因
  const deaths: Death[] = pickWorstPerDimension(runs)
    .map(r => ({
      scenarioId: r.scenario.id, scenarioName: r.scenario.name, dimension: r.scenario.dimension,
      cause: classifyDeath(r.metrics), metrics: r.metrics, narrative: r.scenario.narrative,
    }))
    .filter(d => d.cause !== 'survived');

  // 3) 三风格评分（同一组结果，三种视角）
  const allMetrics = runs.map(r => r.metrics);
  const perStyle = Object.fromEntries(STYLES.map(p => [p.style, scoreStyle(allMetrics, p)])) as Record<StyleName, StyleScore>;

  // 4) 有死因才开处方 + held-out 复测
  let prescription: Prescription | undefined;
  let tradeoff: Tradeoff | undefined;
  if (deaths.length > 0) {
    const profile = getProfile(opts.style);
    prescription = await prescribe(strategy, deaths, treatment, backtest, profile);
    tradeoff = await validateOnHeldOut(strategy, prescription.patchedStrategy, heldOut, backtest, profile);
  }

  return {
    strategyId: strategy.id,
    scenarioSetId: `tx${opts.treatmentSeed}/ho${opts.heldOutSeed}`,
    perStyle, deaths, prescription, tradeoff,
  };
}
```

```ts
import type { Scorecard, Strategy } from '../contracts.ts';

const pct = (v: number) => `${(v * 100).toFixed(1)}%`;
const CAUSE_LABEL: Record<string, string> = {
  liquidation: '强制清算（爆仓）',
  'drawdown-breach': '回撤击穿',
  'stop-loss-bleed': '震荡止损放血',

  }

  L.push('', '## 死因清单');
  if (card.deaths.length === 0) L.push('未发现致死场景。');
  for (const d of card.deaths) {
    L.push(`- 【${d.dimension}】${d.scenarioName} → **${CAUSE_LABEL[d.cause] ?? d.cause}**（回撤 ${pct(d.metrics.maxDrawdownPct)}，收益 ${pct(d.metrics.pnlPct)}）`);
    L.push(`  - ${d.narrative}`);
  }

  if (card.prescription) {
    L.push('', '## 处方');
    L.push(`- 修补项：\`${JSON.stringify(card.prescription.changes)}\``);
    L.push(`- 依据：${card.prescription.rationale}`);
  }
  if (card.tradeoff) {
    L.push('', '## held-out 复测（诚实取舍）');
    L.push(`- 稳健性：风险分 ${card.tradeoff.robustnessGain >= 0 ? '+' : ''}${card.tradeoff.robustnessGain}`);
    L.push(`- 收益代价：平均收益 ${card.tradeoff.returnCost >= 0 ? '+' : ''}${pct(card.tradeoff.returnCost)}`);
    L.push('', '> 本报告不承诺"一键变好"：以上为**未参与治疗**的验证集上的真实取舍。');
  }
  return L.join('\n');
}
```

```ts
// src/redteam/narrate.ts — LLM 死亡报告（增强层），模板兜底（红线：离线可演）
import type { Death } from '../contracts.ts';

const CAUSE_LABEL: Record<Death['cause'], string> = {
  liquidation: '强制清算（爆仓）',
  'drawdown-breach': '回撤击穿风格阈值',
  'stop-loss-bleed': '震荡中反复止损放血',
  survived: '存活',
};

export function fallbackNarrative(death: Death): string {
  const m = death.metrics;
  return `【${death.dimension}】${death.scenarioName}：${death.narrative} ` +
    `体检结论：${CAUSE_LABEL[death.cause]}，最大回撤 ${(m.maxDrawdownPct * 100).toFixed(1)}%，` +
    `期末收益 ${(m.pnlPct * 100).toFixed(1)}%，共 ${m.numTrades} 笔交易。`;
}

/**
 * LLM 增强为显式开关（DOCTOR_LLM_NARRATE=1 且有 key 才启用）：
 * 默认关闭保证 测试/demo 离线确定性；演示时想要生动版再开。任何失败都静默回退模板。
 */
export async function narrateDeath(death: Death): Promise<string> {
  const key = process.env.ANTHROPIC_API_KEY;
  if (process.env.DOCTOR_LLM_NARRATE !== '1' || !key) return fallbackNarrative(death);
  try {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'content-type': 'application/json', 'x-api-key': key, 'anthropic-version': '2023-06-01' },
      body: JSON.stringify({
        model: process.env.DOCTOR_LLM_MODEL ?? 'claude-sonnet-4-6',
        max_tokens: 300,
        messages: [{
          role: 'user',
          content: `你是交易策略法医。用 3 句以内的中文写一段专业死亡报告，必须包含具体数字，不得夸张。\n` +
            `场景：${death.narrative}\n死因分类：${CAUSE_LABEL[death.cause]}\n` +
            `指标：最大回撤 ${(death.metrics.maxDrawdownPct * 100).toFixed(1)}%、收益 ${(death.metrics.pnlPct * 100).toFixed(1)}%、交易 ${death.metrics.numTrades} 笔。`,
        }],
      }),
    });
    if (!res.ok) return fallbackNarrative(death);
    const data = (await res.json()) as { content: { type: string; text?: string }[] };
    const text = data.content.find(b => b.type === 'text')?.text?.trim();
    return text || fallbackNarrative(death);
  } catch {
    return fallbackNarrative(death);
  }
}
```

```ts
// src/backtest/bitget.ts — Bitget Agent Hub 适配器
// 路线 B：用 hub 拉真实历史 K 线，叠加场景冲击，复用 runOnPrices 回测核心。
// 具体子命令/参数以 docs/bitget-hub-notes.md 的 spike 记录为准，仅改本文件，不动契约。
import { execFile } from 'node:child_process';
import { promisify } from 'node:util';
import type { BacktestAdapter, Metrics, Scenario, Strategy } from '../contracts.ts';
import { generatePath } from './path.ts';
import { runOnPrices } from './mock.ts';

const exec = promisify(execFile);

export class BitgetBacktester implements BacktestAdapter {
  async run(strategy: Strategy, scenario: Scenario): Promise<Metrics> {
    // 1) 取真实历史收盘价（命令面以 spike 记录为准）
    const { stdout } = await exec('npx', [
      'bitget-hub', 'candles',
      '--symbol', strategy.universe[0],
      '--timeframe', strategy.timeframe,
      '--limit', '240',
    ], { shell: process.platform === 'win32' });
    const closes: number[] = JSON.parse(stdout).map((c: { close: string | number }) => Number(c.close));

    // 2) 把场景冲击叠加到真实路径尾部 1/3（真实底色 + 对抗剧情）
    const synthetic = generatePath(scenario.shock, closes.length, closes[0]);
    const blended = closes.map((p, i) =>
      i < closes.length * (2 / 3) ? p : p * (synthetic[i] / synthetic[Math.floor(closes.length * (2 / 3))]));

    // 3) 复用同一回测核心，保证 Mock 与真实数据可比
    return runOnPrices(strategy.params, blended);
  }
}
```
